001    package com.studentsonly.grails.plugins.uiperformance;
002    
003    import grails.util.GrailsUtil;
004    import groovy.util.ConfigObject;
005    
006    import java.io.IOException;
007    import java.util.Arrays;
008    import java.util.Collections;
009    import java.util.HashSet;
010    import java.util.List;
011    import java.util.Set;
012    
013    import javax.servlet.Filter;
014    import javax.servlet.FilterChain;
015    import javax.servlet.FilterConfig;
016    import javax.servlet.ServletException;
017    import javax.servlet.ServletRequest;
018    import javax.servlet.ServletResponse;
019    import javax.servlet.http.HttpServletRequest;
020    import javax.servlet.http.HttpServletResponse;
021    
022    import org.codehaus.groovy.grails.commons.ApplicationHolder;
023    import org.codehaus.groovy.grails.commons.ConfigurationHolder;
024    import org.codehaus.groovy.grails.commons.GrailsApplication;
025    import org.springframework.util.AntPathMatcher;
026    
027    /**
028     * Adds 'Expires' and 'Cache-Control' headers plus handles necessary headers when
029     * the resource is gzipped. Only applied when in prod mode and deployed in war.
030     *
031     * @author <a href='mailto:beckwithb@studentsonly.com'>Burt Beckwith</a>
032     */
033    public class CacheFilter implements Filter {
034    
035            private static final long SECONDS_IN_DAY = 60 * 60 * 24;
036            private static final long TEN_YEARS_SECONDS = SECONDS_IN_DAY * 365 * 10;
037            private static final long TEN_YEARS_MILLIS = TEN_YEARS_SECONDS * 1000;
038            private static final String MAX_AGE = "max-age=" + TEN_YEARS_SECONDS;
039            private static final Set<String> EXTENSIONS = new HashSet<String>(
040                            Arrays.asList(new String[] { "js", "css", "gif", "jpg", "png" }));
041    
042            private boolean _production;
043            private boolean _war;
044            private boolean _processImages;
045            private boolean _processCSS;
046            private boolean _processJS;
047            private final AntPathMatcher _pathMatcher = new AntPathMatcher();
048            private List<String> _exclusions;
049    
050            /**
051             * {@inheritDoc}
052             * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse,
053             *      javax.servlet.FilterChain)
054             */
055            public void doFilter(final ServletRequest req, final ServletResponse res, final FilterChain chain)
056                            throws IOException, ServletException {
057    
058                    HttpServletRequest request = (HttpServletRequest)req;
059                    HttpServletResponse response = (HttpServletResponse)res;
060    
061                    String uri = request.getRequestURI();
062    
063                    if (_production && _war && isCacheable(uri)) {
064    
065                            response.setDateHeader("Expires", System.currentTimeMillis() + TEN_YEARS_MILLIS);
066                            response.setHeader("Cache-Control", MAX_AGE);
067                            if (uri.endsWith(".gz.css") || uri.endsWith(".gz.js")) {
068                                    response.addHeader("Content-Encoding", "gzip");
069                            }
070                    }
071    
072                    chain.doFilter(request, response);
073            }
074    
075            private boolean isCacheable(final String uri) {
076    
077                    if (!uri.contains("__v")) {
078                            return false;
079                    }
080    
081                    if (isExcluded(uri)) {
082                            return false;
083                    }
084    
085                    int index = uri.lastIndexOf('.');
086                    if (index == -1) {
087                            return false;
088                    }
089    
090                    String extension = uri.substring(index + 1).toLowerCase();
091                    if (!EXTENSIONS.contains(extension)) {
092                            return false;
093                    }
094    
095                    if (extension.equals("css")) {
096                            if (!_processCSS) {
097                                    return false;
098                            }
099                    }
100                    else if (extension.equals("js")) {
101                            if (!_processJS) {
102                                    return false;
103                            }
104                    }
105                    else if (!_processImages) {
106                            return false;
107                    }
108    
109                    return true;
110            }
111    
112            private boolean isExcluded(final String uri) {
113                    for (String pattern : _exclusions) {
114                            if (_pathMatcher.match(pattern, uri)) {
115                                    return true;
116                            }
117                    }
118                    return false;
119            }
120    
121            /**
122             * {@inheritDoc}
123             * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
124             */
125            public void init(final FilterConfig filterConfig) {
126                    _production = GrailsApplication.ENV_PRODUCTION.equals(GrailsUtil.getEnvironment());
127                    _war = ApplicationHolder.getApplication().isWarDeployed();
128                    
129                    _processImages = getConfigBoolean("processImages");
130                    _processCSS = getConfigBoolean("processCSS");
131                    _processJS = getConfigBoolean("processJS");
132    
133                    findExclusions();
134            }
135    
136            private boolean getConfigBoolean(final String name) {
137                    Boolean value = (Boolean)getConfigProperty(name);
138                    return value == null ? true : value;
139            }
140    
141            @SuppressWarnings("unchecked")
142            private void findExclusions() {
143                    _exclusions = (List<String>)getConfigProperty("exclusions");
144                    if (_exclusions == null) {
145                            _exclusions = Collections.emptyList();
146                    }
147            }
148    
149            private Object getConfigProperty(final String name) {
150    
151                    ConfigObject current = ConfigurationHolder.getConfig();
152                    String[] parts = ("uiperformance." + name).split("\\.");
153    
154                    // navigate down to the last element, returning null if any level isn't
155                    // a ConfigObject or doesn't exist (don't auto-create like ConfigObject does).
156                    for (int i = 0; i < parts.length - 1; i++) {
157                            Object thing = current.get(parts[i]);
158                            if (thing instanceof ConfigObject) {
159                                    current = (ConfigObject)thing;
160                            }
161                            else {
162                                    return null;
163                            }
164                    }
165    
166                    return current.get(parts[parts.length - 1]);
167            }
168            
169            /**
170             * {@inheritDoc}
171             * @see javax.servlet.Filter#destroy()
172             */
173            public void destroy() {
174                    // nothing to do
175            }
176    }
